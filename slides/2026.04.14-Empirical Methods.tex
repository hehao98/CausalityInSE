% ---------- Slide 5 ----------------------------------------------------------
\begin{frame}{The fundamental problem: \redt{No counterfactuals!}}
\begin{itemize}\setlength\itemsep{0.3em}
  \item Mathematically speaking:
    \begin{itemize}\setlength\itemsep{0.2em}
      \item Let $Y_i(1)$ be the outcome on $i$ when $i$ gets the intervention
      \item Let $Y_i(0)$ be the outcome on $i$ when $i$ does not get the intervention
      \item The true causal effect of \redt{intervening on $i$} is $Y_i(1) - Y_i(0)$
      \item However, for each $i$, \textbf{you either see $Y_i(1)$ or $Y_i(0)$, but never both!}
    \end{itemize}
  \item The workaround (attributing to the experiment pioneers in 1930s)
    \begin{itemize}\setlength\itemsep{0.2em}
      \item We \textbf{estimate} population level quantities like the average treatment effect (ATE)
      \[
        ATE = E[Y_i(1) - Y_i(0) \mid i \in P] = E[Y_i(1) \mid i \in P] - E[Y_i(0) \mid i \notin P]
      \]
      \item $P$ is the population of interest defined by the researcher (e.g., all human)
    \end{itemize}
\end{itemize}
\end{frame}

% ---------- Slide 7 ----------------------------------------------------------
\begin{frame}{Decomposing \textbf{selection bias} --- And how to get rid of it?}

\vspace{-0.2em}
\begin{itemize}\setlength\itemsep{0.25em}
  \item \textbf{Selection bias}: CMU students would earn more \redt{even without CMU}
    \begin{itemize}
      \item Higher motivation, wealth, academic ability $\Rightarrow$ na\"ive 3$\times$ gap is inflated
    \end{itemize}
  \item \textbf{Mathematical decomposition}: Let $D_i\in\{0,1\}$ be whether unit $i$ is treated (e.g.\ attends CMU).
  The \textbf{na\"ive comparison} we are tempted to make:
  \[
    \underbrace{E[Y_i\mid D_i{=}1] - E[Y_i\mid D_i{=}0]}_{\text{na\"ive estimator}}
    \;=\; E[Y_i(1)\mid D_i{=}1] \;-\; E[Y_i(0)\mid D_i{=}0]
  \]

  \vspace{-0.2em}
  \noindent Add and subtract $E[Y_i(0)\mid D_i{=}1]$ (the \emph{unobserved} baseline of treated units):
  \[
    =\;\underbrace{E[Y_i(1)-Y_i(0)\mid D_i{=}1]}_{\text{ATT (true causal effect)}}
    \;+\;
    \underbrace{E[Y_i(0)\mid D_i{=}1] - E[Y_i(0)\mid D_i{=}0]}_{\redt{\text{selection bias}}}
  \]
  \item \textbf{To eliminate selection bias}: We need $E[Y_i(0)\mid D_i{=}1] = E[Y_i(0)\mid D_i{=}0]$
\end{itemize}

\end{frame}

% ---------- Slide 8 ----------------------------------------------------------
\begin{frame}{\redt{Randomization} + \redt{big numbers} eliminates selection bias}

\textbf{Randomization}: Assign treatment $D_i$ by a coin flip, \emph{independently} of everything in $i$:
\[
  D_i \;\perp\!\!\!\perp\; \bigl(Y_i(0),\, Y_i(1)\bigr)
\]

\vspace{-0.5em}
\noindent This independence kills selection bias immediately:
\[
  E[Y_i(0)\mid D_i{=}1] \;=\; E[Y_i(0)\mid D_i{=}0] \;=\; E[Y_i(0)]
\]

\vspace{-0.5em}
\noindent So the na\"ive estimator \emph{becomes} the ATE:
\[
  \underbrace{E[Y_i\mid D_i{=}1] - E[Y_i\mid D_i{=}0]}_{\text{na\"ive estimator}}
  \;=\; \underbrace{E[Y_i(1)-Y_i(0)]}_{\text{ATE}} \;+\; \underbrace{0}_{\text{bias} = 0}
\]

\vspace{-0.5em}
\begin{itemize}\setlength\itemsep{0.25em}
  \item \textbf{Big numbers}: sample averages $\hat{E}[\cdot] \xrightarrow{n\to\infty} E[\cdot]$ by the law of large numbers
  \item With enough units, $\bar{Y}_{\text{treated}} - \bar{Y}_{\text{control}}$ reliably estimates ATE (Fisher, Neyman, 1930s)
\end{itemize}

\end{frame}

% ---------- Slide 11 ---------------------------------------------------------
\begin{frame}{Can I interpret the regression coefficient as the causal effect?}

\textbf{Intuition}: OLS implicitly compares units \emph{with similar covariates}.\\[0.3em]
The coefficient $\hat\beta$ on $D_i$ is a \redt{weighted average of within-group comparisons}
after ``partialing out'' the controls $X_i$ (Frisch--Waugh--Lovell theorem, 1933, 1963)

\vspace{0.6em}
$\hat\beta$ \textbf{has a causal interpretation} when the following \redt{four conditions} hold:
\begin{enumerate}\setlength\itemsep{0.3em}
  \item \textbf{No omitted confounders} --- every variable that affects \emph{both} $D_i$ and $Y_i$ is
    included in $X_i$.  Otherwise, \redt{omitted-variable bias} (OVB) occurs.
  \item \textbf{No bad controls} --- you have \emph{not} conditioned on colliders or post-treatment mediators, which can \redt{create or mask} associations.
  \item \textbf{Correct functional form} --- $E[Y_i \mid D_i, X_i]$ is approximately linear; if the true relationship has strong nonlinearities or interactions, $\hat\beta$ may not recover the ATE.
  \item \textbf{No reverse causality} --- The outcome $Y_i$ must not feed back into treatment $D_i$.
\end{enumerate}

\vspace{-0.2em}
\begin{footnotesize}See Angrist \& Pischke, \textit{Mostly Harmless Econometrics} (2009), Ch.\ 3 for more details.\end{footnotesize}

\end{frame}

% ---------- Slide 12 ---------------------------------------------------------
\begin{frame}{The conditional ignorability assumption (CIA)}

\textbf{CIA} (a.k.a.\ \emph{unconfoundedness}, \emph{selection on observables}):
\[
  \bigl\{Y_i(0),\,Y_i(1)\bigr\} \;\perp\!\!\!\perp\; D_i \;\Big|\; X_i
\]
\textit{Conditional on observed covariates $X_i$, treatment assignment is as good as random.}

\vspace{0.3em}
\textbf{Key consequence} (MHE Ch.\ 3, \S3.3): when CIA + \emph{common support} ($0 < P(D_i{=}1\mid X_i) < 1$) hold,
\[
  \underbrace{E\bigl[Y_i(1) - Y_i(0)\bigr]}_{\text{ATE}}
  \;=\;
  E\!\Bigl[\underbrace{E[Y_i \mid D_i{=}1,\, X_i] - E[Y_i \mid D_i{=}0,\, X_i]}_{\text{within-group comparison at each }X_i}\Bigr]
\]

\begin{itemize}\setlength\itemsep{0.25em}
  \item \textbf{Intuition}: at every value $X_i = x$, CIA makes $D_i$ random $\Rightarrow$ within-cell comparison is unbiased $\Rightarrow$ average over $x$ gives ATE
  \item \redt{CIA is the key identifying assumption} behind OLS, matching, IPW, and most observational estimators
  \item \textbf{Reference}: Angrist \& Pischke, \textit{Mostly Harmless Econometrics} (2009), Ch.\ 3 (esp.\ \S3.2--3.3)
\end{itemize}

\end{frame}
\end{comment}

\begin{frame}{How do I know that the above conditions hold?}

\textbf{Key insight}: The four conditions \redt{cannot be verified from data alone}.
You need \textbf{theory} --- substantive knowledge about \emph{how} variables are causally related.

\vspace{0.5em}
\textbf{Directed Acyclic Graphs (DAGs)} are a formal language for encoding causal theory
\hfill{\small (Pearl, \textit{Causality}, 2009; Spirtes et al., \textit{Causation, Prediction, and Search}, 2000)}

\vspace{-0.3em}
\begin{center}
\begin{tikzpicture}[
    node distance=0.8cm and 1.0cm,
    every node/.style={font=\footnotesize},
    dagnode/.style={draw, circle, minimum size=0.6cm, inner sep=1pt},
    arr/.style={-{Stealth[length=4pt]}, semithick}
  ]
  % --- Confounder (fork) ---
  \node[dagnode] (X1) {$X$};
  \node[dagnode, below left=of X1] (D1) {$D$};
  \node[dagnode, below right=of X1] (Y1) {$Y$};
  \draw[arr] (X1) -- (D1);
  \draw[arr] (X1) -- (Y1);
  \draw[arr] (D1) -- (Y1);
  \node[below=0.2cm of $(D1)!0.5!(Y1)$, font=\footnotesize\bfseries] {Confounder};
  \node[below=0.55cm of $(D1)!0.5!(Y1)$, font=\scriptsize, text=emphred] {Must control for $X$};

  % --- Mediator (chain) ---
  \node[dagnode, right=2.8cm of X1] (D2) {$D$};
  \node[dagnode, right=0.9cm of D2] (M2) {$M$};
  \node[dagnode, right=0.9cm of M2] (Y2) {$Y$};
  \draw[arr] (D2) -- (M2);
  \draw[arr] (M2) -- (Y2);
  \node[below=0.6cm of M2, font=\footnotesize\bfseries] {Mediator};
  \node[below=0.95cm of M2, font=\scriptsize, text=emphred, align=center] {Do \emph{not} control for $M$\\[-1pt](blocks the causal path)};

  % --- Collider ---
  \node[dagnode, right=2.8cm of M2] (D3) {$D$};
  \node[dagnode, right=1.2cm of D3] (Y3) {$Y$};
  \node[dagnode, below=0.9cm of $(D3)!0.5!(Y3)$] (C3) {$C$};
  \draw[arr] (D3) -- (Y3);
  \draw[arr] (D3) -- (C3);
  \draw[arr] (Y3) -- (C3);
  \node[below=0.2cm of C3, font=\footnotesize\bfseries] {Collider};
  \node[below=0.55cm of C3, font=\scriptsize, text=emphred, align=center] {Do \emph{not} control for $C$\\[-1pt](opens a spurious path)};
\end{tikzpicture}
\end{center}

\vspace{-1em}
\begin{itemize}
  \item A DAG tells you \textbf{which variables to control for} and which to leave alone
  \item The \textbf{backdoor criterion} (Pearl, 1995): block all non-causal paths from $D$ to $Y$
\end{itemize}

\end{frame}

% ---------- Slide 14 ---------------------------------------------------------
\begin{frame}{Practical considerations and limitations}

\textbf{DAGs are useful for\ldots}
\begin{itemize}\setlength\itemsep{0.15em}
  \item \textbf{Theory proposition} --- force you to spell out \emph{every} causal pathway you believe
  \item \textbf{Iterative refinement} --- collaborators can critique, extend, or prune edges
  \item \textbf{Making assumptions transparent} --- reviewers can see exactly what you assumed
\end{itemize}

\vspace{0.5em}
\textbf{But DAGs encode \redt{researcher beliefs}, which are fundamentally untestable.}\\[0.2em]
Two researchers with different DAGs $\Rightarrow$ different conclusions from the same data.

\vspace{0.5em}
\textbf{Regression (and matching / IPW) share deeper limitations:}
\begin{itemize}\setlength\itemsep{0.15em}
  \item \textbf{Unobservable confounders} --- some variables are \emph{inherently} unmeasurable
    (e.g.\ ``ability,'' ``motivation,'' ``organization culture'')
  \item \textbf{Reverse causality} --- when future outcomes leak into treatment,
    conditioning on observables does not help;
    the causal direction itself is ambiguous
\end{itemize}
\end{frame}

  \begin{itemize}\setlength\itemsep{0.2em}
    \item Major US orchestras adopted \textbf{blind auditions} (a physical screen hides the performer) at different times
    \item Compares hiring of women \emph{before vs.\ after} the screen was introduced, relative to orchestras that did not yet adopt it
  \end{itemize}

  \vspace{0.4em}
  \textbf{Finding}:
  \begin{itemize}\setlength\itemsep{0.2em}
    \item Blind auditions increased the probability of women advancing by \textbf{$\sim$25--46\%}
  \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{What else is out there?}

\textbf{A.\ ``I don't care about averages. Tell me \redt{for whom} the effect holds.''}
\begin{itemize}\setlength\itemsep{0.15em}
  \item Use ML to estimate \textbf{heterogeneous treatment effects} across individuals or subgroups (i.e., find who benefits the most).
  \item Examples: \textbf{causal forests} (Wager \& Athey, 2018), \textbf{double/debiased ML} (Chernozhukov et al., 2018)
\end{itemize}

\vspace{0.3em}
\textbf{B.\ ``\redt{How wrong} would the assumptions need to be to overturn the result?''}
\begin{itemize}\setlength\itemsep{0.15em}
  \item \textbf{Rosenbaum bounds}: quantify how large an unmeasured confounder must be to flip a significant finding --- a stress test for observational results
  \item \textbf{Oster (2019)}: bounds selection on unobservables by how much $\hat\beta$ moves as you add controls and how $R^2$ changes
\end{itemize}

\vspace{0.3em}
\textbf{\redt{Important}}: these \textbf{supplement, not replace}, the identification strategies above.
You still need a credible design so that ML and sensitivity analysis \redt{make sense}.
\end{frame}

\begin{frame}{Recap}
\begin{enumerate}\setlength\itemsep{0.4em}
  \item Motivation: The intervention-outcome problem
  \item The fundamental problem of causal inference: No counterfactuals
  \item Selection bias \& why randomization helps (but isn't always feasible)
  \item What takes for a regression on observational data to be causal
    \begin{itemize}
      \item Conditional ignorability: The formal assumption for causal interpretation
      \item DAGs: confounders, colliders, and mediators
    \end{itemize}
  \item Design-based identification (RDD, DiD, IV, panel, etc.)
  \item Examples: College education on income, gender bias on high-skilled jobs, AI coding 
  tools on development outcomes
  \item What else is out there? ML-based discovery and sensitivity analysis
\end{enumerate}
\end{frame}
